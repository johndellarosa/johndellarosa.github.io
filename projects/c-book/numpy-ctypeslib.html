Instead of accepting a raw ctypes.POINTER, ndpointer lets you enforce at call-time that an argument is:
  • A numpy array
  • Of a specific dtype (e.g., np.double, np.int32)
  • With a given number of dimensions (e.g., ndim=1 for a 1D array)
  • Having particular memory flags (e.g., flags="C_CONTIGUOUS" for C-contiguous arrays)
                
                from numpy.ctypeslib import ndpointer
                import ctypes
                import numpy as np

                lib = ctypes.CDLL("libvector_ops.so")

                # C signature: void add_arrays(double *a, double *b, double *out, size_t n);
                lib.add_arrays.argtypes = (
                    ndpointer(dtype=np.double, ndim=1, flags="C_CONTIGUOUS,ALIGNED"),
                    ndpointer(dtype=np.double, ndim=1, flags="C_CONTIGUOUS,ALIGNED"),
                    ndpointer(dtype=np.double, ndim=1, flags="C_CONTIGUOUS,ALIGNED,WRITEABLE"),
                    ctypes.c_size_t,
                )
                lib.add_arrays.restype = None
