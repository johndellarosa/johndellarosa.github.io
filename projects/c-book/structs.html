Structs are a way of grouping related data together into one object, similar to classes in Python (and C++).

Let's say that we have struct defined in C as follows: 

                    struct Point {
                        int x;
                        int y;
                    };
We can represent this struct in Python using ctypes as follows: 

                    from ctypes import Structure, c_int

                    class Point(Structure):
                        _fields_ = [("x", c_int), ("y", c_int)]
                
                p = Point(10, 20)

Passing Structs to and from C Functions

Let's expand our C code to include a function which takes a struct as an argument: 

                    #include <stdio.h>

                    struct Point {
                        int x;
                        int y;
                    };

                    void modify_point(struct Point* p, int new_x, int new_y) {
                        p->x = new_x;
                        p->y = new_y;
                    }
Compile this code into a DLL, which we will call my_struct.dll. 

                    import ctypes

                    # Define the struct using ctypes
                    class Point(ctypes.Structure):
                        _fields_ = [("x", ctypes.c_int),
                                    ("y", ctypes.c_int)]

                    # Load the C library
                    my_functions = ctypes.CDLL(r'.\my_struct.dll')  # Adjust the path as necessary

                    # Set up the function argument and return types
                    my_functions.modify_point.argtypes = [ctypes.POINTER(Point), ctypes.c_int, ctypes.c_int]
                    my_functions.modify_point.restype = None

                    # Create an instance of the struct
                    p = Point(10, 20)
                    print(f"Before modify_point: ({p.x}, {p.y})")

                    # Call the C function
                    my_functions.modify_point(ctypes.byref(p), 25, 30)

                    # Print the modified values
                    print(f"After modify_point: ({p.x}, {p.y})")

Structs with Arrays and Pointers


                    struct ArrayPoint {
                        int x;
                        int y;
                        int data[5];  // Array of 5 integers
                    };

                    // Function to modify the array inside the struct
                    void modify_array(struct ArrayPoint* ap) {
                        for (int i = 0; i < 5; i++) {
                            ap->data[i] *= 2;  // Multiply each element of the array by 2
                        }
                    }
                

                    import ctypes


                    class ArrayPoint(ctypes.Structure):
                        _fields_ = [("x", ctypes.c_int),
                                    ("y", ctypes.c_int),
                                    ("data", ctypes.c_int * 5)]  # Array of 5 integers

                    my_functions = ctypes.CDLL(r'.\my_struct_2.dll')  # Adjust the path as necessary
                    my_functions.modify_array.argtypes = [ctypes.POINTER(ArrayPoint)]
                    my_functions.modify_array.restype = None

                    # Create an instance of the struct
                    ap = ArrayPoint(10, 20, (1, 2, 3, 4, 5))

                    # Print original data
                    print(f"Original ArrayPoint data: {ap.data[:]}")

                    # Call the C function to modify the array
                    my_functions.modify_array(ctypes.byref(ap))

                    # Print the modified data
                    print(f"Modified ArrayPoint data: {ap.data[:]}")
